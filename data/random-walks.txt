# Random Walks

[RED :0]   255   0   0 rgb
[ORANGE:0] 255 165   0 rgb
[BLUE :0]    0   0 255 rgb

512 @win-size

win-size 1.1 * dup window

[rnd-walk:0] win-size -0.5 * 0 move    win-size 2 * { rnd rnd 5 * -2.5 + rline } loop

# draw horizontal line
win-size { 3 * win-size -2 / + 0 point } loop

RED    rnd-walk
BLUE   rnd-walk
ORANGE rnd-walk
